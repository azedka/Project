import random
import string


def generer_mot_de_passe():
    know_letter = input("Voulez-vous des lettres dans votre mot de passe? (oui/non) : ").lower()

    if know_letter == "oui":
        lettre = int(input("Combien de lettres voulez-vous dans votre mot de passe : "))
        random_letters = [random.choice(string.ascii_letters) for _ in range(lettre)]

        know_chiffre = input("Voulez-vous des chiffres dans votre mot de passe? (oui/non) : ").lower()

        if know_chiffre == "oui":
            nombre = int(input("Combien de chiffres voulez-vous dans votre mot de passe : "))
            random_digits = [random.randint(0, 9) for _ in range(nombre)]
        else:
            random_digits = []

        random_strings = [''.join(random_letters + [str(digit) for digit in random_digits])]
        print("Chaînes aléatoires:", random_strings)
    else:
        print("Vous devez spécifier si vous voulez des lettres dans votre mot de passe.")


# Fonction pour demander à l'utilisateur s'il veut rerun le programme
def rerun():
    return input("Voulez-vous générer un autre mot de passe? (oui/non)").lower() == "oui"


# Boucle principale
while True:
    generer_mot_de_passe()

    # Demander à l'utilisateur s'il veut rerun
    if not rerun():
        break
