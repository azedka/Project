import random
print("                               LE JEU DU PIERRE, FEUILLE, CISEAU ")
print("                                     voici les r√®gles :")
print("                                   La feuille bat la pierre")
print("                                   La pierre bat les ciseaux")
print("                                  Les ciseaux battent la feuille")
score_bot = 0
score_joueur = 0
def jeu():
    global score_bot
    global score_joueur
    comp_choice = random.randint(1, 3)
    user_choice = int(input("Veuillez incerer ce que vous allez jouer (feuille = 1/ pierre = 2/ ciseaux = 3) : "))
    if (comp_choice == 1) and (user_choice == 1):
        print("IL Y A UNE EGALITE (feuille-feuille)")

    if (comp_choice == 1) and (user_choice == 2):
        print("L'ORDINATEUR A GAGNE (feuille-pierre)")
        score_bot += 1

    if (comp_choice == 1) and (user_choice == 3):
        print("VOUS AVEZ GAGNE (feuille-ciseaux)")
        score_joueur += 1

    if (comp_choice == 2) and (user_choice == 1):
        print("VOUS AVEZ GAGNE (pierre - feuille)")
        score_joueur += 1

    if (comp_choice == 2) and (user_choice == 2):
        print("IL Y A UNE EGALITE (pierre-pierre)")

    if (comp_choice == 2) and (user_choice == 3):
        print("L'ORDINATEUR A GAGNE (pierre-ciseaux)")
        score_bot += 1

    if (comp_choice == 3) and (user_choice == 1):
        print("L'ORDINATEUR A GAGNE (ciseaux-feuille)")
        score_bot += 1

    if (comp_choice == 3) and (user_choice == 2):
        print("VOUS AVEZ GAGNE (ciseaux-pierre)")
        score_joueur += 1

    if (comp_choice == 3) and (user_choice == 3):
        print("IL Y A UNE EGALITE (ciseaux-ciseaux)")

while True:
        jeu()
        print("Il y a", score_bot, "-", score_joueur)
